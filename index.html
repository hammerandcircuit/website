function setLanguage(lang) { currentLang = lang; document.querySelectorAll('.lang-btn').forEach(btn => { btn.classList.remove('bg-tool-red', 'text-white'); btn.classList.add('bg-black', 'text-gray-300'); }); const langBtn = document.getElementById(`lang-${lang}`); if (langBtn) { langBtn.classList.remove('bg-black', 'text-gray-300'); langBtn.classList.add('bg-tool-red', 'text-white'); } document.querySelectorAll('[data-i18n]').forEach(el => { const key = el.getAttribute('data-i18n'); if (translations[lang] && translations[lang][key]) el.innerText = translations[lang][key]; }); const activeNav = document.querySelector('.switch-active'); const activePage = activeNav ? activeNav.id.replace('nav-', '') : 'home'; renderPage(activePage); }
